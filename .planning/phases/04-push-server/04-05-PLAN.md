    ```dockerfile
    # --- Build stage ---
    FROM elixir:1.18-alpine AS builder

    RUN apk add --no-cache build-base git

    WORKDIR /app

    ENV MIX_ENV=prod

    RUN mix local.hex --force && mix local.rebar --force

    COPY mix.exs mix.lock ./
    RUN mix deps.get --only prod
    RUN mix deps.compile

    COPY config config
    COPY lib lib

    RUN mix release

    # --- Runtime stage ---
    FROM alpine:3.21 AS app

    RUN apk add --no-cache libstdc++ openssl ncurses-libs

    WORKDIR /app

    COPY --from=builder /app/_build/prod/rel/push_server ./

    EXPOSE 4000

    CMD ["bin/push_server", "start"]
    ```

<task type="checkpoint:human-verify" gate="blocking">
  <what-built>
    Complete push-server: Phoenix Channel layer, Broadway Kafka consumer, PubSub Redis adapter, PromEx metrics, Docker image, docker-compose service block.
  </what-built>
  <how-to-verify>
    Run these steps in order:

    **Step 1 — Start the full stack:**
    ```bash
    cd /Users/iedo/Study/mototaxi
    docker compose up --build -d
    # Wait ~30s for redpanda-init to finish topic creation
    docker compose logs push-server --tail=30
    ```
    Expected: push-server logs show "Running PushServerWeb.Endpoint with cowboy" on port 4000. No crash loops.

    **Step 2 — Verify /metrics endpoint:**
    ```bash
    curl -s http://localhost:4000/metrics | grep push_server_
    ```
    Expected: Lines containing `push_server_connections_active`, `push_server_messages_delivered_total`, `push_server_delivery_latency_ms`.

    **Step 3 — Connect a WebSocket client and join a channel:**
    Install wscat if needed: `npm install -g wscat`
    ```bash
    wscat -c "ws://localhost:4000/socket/websocket?vsn=2.0.0"
    ```
    Once connected, send a join message:
    ```json
    ["1","1","customer:customer-1","phx_join",{}]
    ```
    Expected response: A `phx_reply` with status "ok". If the simulator is running and customer-1 has a driver assignment in Redis, you should also receive a `location_update` event with the initial position.

    **Step 4 — Verify ongoing updates (requires simulator running):**
    Keep the wscat connection open. Within 1-2 seconds, you should receive additional `location_update` events as the simulator emits position updates.
    Expected payload shape:
    ```json
    {"driver_id":"driver-1","lat":-23.52,"lng":-46.63,"bearing":142.5,"speed_kmh":34.2,"emitted_at":"2026-..."}
    ```

    **Step 5 — Verify cross-replica delivery (PUSH-04):**
    ```bash
    docker compose up --scale push-server=2 -d
    ```
    Connect wscat to replica 1 (port 4000) and check docker compose logs to see which replica consumes the Kafka message. Both replicas should deliver the update to their respective connected clients.

    **Step 6 — Error path verification:**
    ```bash
    wscat -c "ws://localhost:4000/socket/websocket?vsn=2.0.0"
    # Send join for non-existent customer:
    ["1","1","customer:nonexistent-999","phx_join",{}]
    ```
    Expected: `phx_reply` with status "error" and reason "unknown_customer".
  </how-to-verify>
  <resume-signal>Type "approved" if all 6 steps pass, or describe which steps failed and what was observed.</resume-signal>
</task>
